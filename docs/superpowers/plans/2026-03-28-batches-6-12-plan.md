**Shared Audio Simulation Pattern:** All Batch 6 demos simulate audio data on CPU using multi-frequency sine waves. No Web Audio API. Pattern:

```tsx
// In useFrame callback — simulate 64-band frequency spectrum
const bands = useRef(new Float32Array(64));
useFrame((_, delta) => {
  const t = performance.now() * 0.001;
  for (let i = 0; i < 64; i++) {
    const freq = 1 + i * 0.5;
    const amp = 1.0 / (1 + i * 0.3); // bass-heavy falloff
    bands.current[i] = (Math.sin(t * freq) * 0.5 + 0.5) * amp
      + Math.sin(t * freq * 1.7 + i) * amp * 0.3; // harmonic
  }
});
```

const beat = Math.pow(Math.sin(t * Math.PI * 2 * bpm / 60) * 0.5 + 0.5, 8); // sharp pulse

### Demo 1: beat-pulse-grid

- [ ] **Step 1: Create `src/demos/beat-pulse-grid/index.tsx`**

20x20 grid (400) instanced boxes. Heights driven by simulated beat frequency. Technique combo: instanced mesh + CPU-driven audio uniforms + hash-based per-instance variation + multi-stop color gradient.

```tsx
import { useRef, useMemo } from 'react';
import { useFrame } from '@react-three/fiber';
import * as THREE from 'three/webgpu';
import {
  color, time, float, vec3, hash, mix, smoothstep,
  positionWorld, positionLocal, normalLocal, oscSine,
  Fn, instanceIndex,
} from 'three/tsl';

const GRID = 20;
const COUNT = GRID * GRID;

export default function BeatPulseGrid() {
  const meshRef = useRef<THREE.InstancedMesh>(null);
  const heightsRef = useRef(new Float32Array(COUNT));

  // Place instances in grid
  useMemo(() => {
    const dummy = new THREE.Object3D();
    const mesh = meshRef.current;
    if (!mesh) return;
    for (let i = 0; i < COUNT; i++) {
      const x = (i % GRID) - GRID / 2 + 0.5;
      const z = Math.floor(i / GRID) - GRID / 2 + 0.5;
      dummy.position.set(x * 0.55, 0, z * 0.55);
      dummy.scale.set(0.25, 0.1, 0.25);
      dummy.updateMatrix();
      mesh.setMatrixAt(i, dummy.matrix);
    }
    mesh.instanceMatrix.needsUpdate = true;
  }, []);

  // Material: height-based color gradient (blue -> cyan -> magenta -> white)
  const material = useMemo(() => {
    const mat = new THREE.MeshStandardNodeMaterial();
    const h = positionWorld.y.div(3.0).clamp(0, 1);
    const c1 = color(0x1122aa); // blue
    const c2 = color(0x22cccc); // cyan
    const c3 = color(0xff44aa); // magenta
    const c4 = color(0xffffff); // white
    const grad = mix(mix(c1, c2, smoothstep(0.0, 0.3, h)),
      mix(c3, c4, smoothstep(0.5, 1.0, h)),
      smoothstep(0.25, 0.6, h));
    mat.colorNode = grad;
    mat.emissiveNode = grad.mul(float(1.5).add(h.mul(2.0)));
    mat.roughness = 0.3;
    mat.metalness = 0.2;
    return mat;
  }, []);

  useFrame((_, delta) => {
    const mesh = meshRef.current;
    if (!mesh) return;
    const t = performance.now() * 0.001;

    // Simulate beat pulses at different BPMs for variety
    const dummy = new THREE.Object3D();
    for (let i = 0; i < COUNT; i++) {
      const x = (i % GRID) - GRID / 2 + 0.5;
      const z = Math.floor(i / GRID) - GRID / 2 + 0.5;
      const dist = Math.sqrt(x * x + z * z) / (GRID / 2);

      // Multi-frequency beat simulation
      const bass = Math.pow(Math.sin(t * Math.PI * 2 * 120 / 60) * 0.5 + 0.5, 8);
      const mid = Math.pow(Math.sin(t * Math.PI * 2 * 130 / 60 + dist * 3) * 0.5 + 0.5, 6);
      const hi = Math.pow(Math.sin(t * Math.PI * 2 * 140 / 60 + dist * 6) * 0.5 + 0.5, 4);

      const bandIdx = Math.floor(dist * 3) % 3;
      const intensity = bandIdx === 0 ? bass : bandIdx === 1 ? mid : hi;
      const h = intensity * 3.0 + 0.1;

      dummy.position.set(x * 0.55, h / 2, z * 0.55);
      dummy.scale.set(0.25, h, 0.25);
      dummy.updateMatrix();
      mesh.setMatrixAt(i, dummy.matrix);
    }
    mesh.instanceMatrix.needsUpdate = true;
  });

  return (
    <>
      <ambientLight intensity={0.05} />
      <directionalLight position={[5, 10, 5]} intensity={0.3} />
      <pointLight position={[0, 4, 0]} intensity={8} color="#ff2266" distance={15} />
      <pointLight position={[-4, 2, -4]} intensity={5} color="#2266ff" distance={12} />
      <pointLight position={[4, 2, 4]} intensity={5} color="#22cccc" distance={12} />

      <instancedMesh ref={meshRef} args={[undefined, undefined, COUNT]} material={material} frustumCulled={false}>
        <boxGeometry args={[1, 1, 1]} />
      </instancedMesh>
    </>
  );
}
```
